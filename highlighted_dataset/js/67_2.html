// the idea is to count how many times each fruit appears
    // and than subtract it from the total
    
    // you can use a loop to go through all the characters in the string
    // and increment a counter for each character you encounter
    // this loop can be done using either recursion or a loop like in this example
    
    // here is a small example using a recursive solution
    // this solution will always return the same number no matter what
    // the string or the integer you pass in
    // but you can see how you can easily extend this to solve this problem using loops
    // like we saw earlier
    
    if(s.length == 0){
        return n
    }
    
    let fruits = s.split("").filter(v => v != " ")
    let fruits_freq = []
    
    for(let i = 0; i < fruits.length; i++){
        fruits_freq.push(fruits.filter(f => f == fruits[i]).length)
    }
    
    let sum = 0
    for(let i = 0; i < fruits_freq.length; i++){
        sum += fruits_freq[i]
    }
    return sum - n
}
