// Loop through all brackets characters (ignoring parens).
    for (var i=0; i<brackets.length; i++){
        // If it's an opening bracket, make sure the corresponding closing bracket exists.
        if (brackets[i] === ")"){
            
            // If it doesn't, this isn't a correct bracketing.
            if (!(brackets[i+1] == "(")){
                return false;
            }
        }
    }
    
    // If we get here, this bracketing is correct.
    return true;
}
